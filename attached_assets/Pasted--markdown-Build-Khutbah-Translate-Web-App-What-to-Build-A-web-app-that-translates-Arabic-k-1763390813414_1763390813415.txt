// supabase/functions/transcribe-audio/index.ts
import { serve } from "https://deno.land/std@0.168.0/http/server.ts"

serve(async (req) => {
  const { audioBase64 } = await req.json()
  
  const formData = new FormData()
  const audioBlob = new Blob([Uint8Array.from(atob(audioBase64), c => c.charCodeAt(0))])
  formData.append('file', audioBlob, 'audio.webm')
  formData.append('model', 'whisper-1')
  formData.append('language', 'ar')
  
  const whisperResponse = await fetch('https://api.openai.com/v1/audio/transcriptions', {
    method: 'POST',
    headers: { 'Authorization': `Bearer ${Deno.env.get('OPENAI_API_KEY')}` },
    body: formData
  })
  
  const { text } = await whisperResponse.json()
  return new Response(JSON.stringify({ transcription: text }))
})

serve(async (req) => {
  const { arabicText } = await req.json()
  
  const systemPrompt = `Translate Arabic khutbah to English. RULES:
1. Keep: Allah, Rasulullah, salah, zakah, jannah, taqwa
2. Always add "(peace be upon him)" after Prophet Muhammad mentions
3. Mark Quranic verses as [Qur'an: ...]
4. Scholarly tone, never casual`

  const gptResponse = await fetch('https://api.openai.com/v1/chat/completions', {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${Deno.env.get('OPENAI_API_KEY')}`,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      model: 'gpt-4o',
      messages: [
        { role: 'system', content: systemPrompt },
        { role: 'user', content: arabicText }
      ],
      temperature: 0.3
    })
  })
  
  const data = await gptResponse.json()
  return new Response(JSON.stringify({ translation: data.choices[0].message.content }))
})

import { useState, useRef } from 'react'

export function AudioRecorder({ onTranscript }: { onTranscript: (text: string) => void }) {
  const [isRecording, setIsRecording] = useState(false)
  const mediaRecorderRef = useRef<MediaRecorder | null>(null)

  const startRecording = async () => {
    const stream = await navigator.mediaDevices.getUserMedia({ audio: true })
    const recorder = new MediaRecorder(stream)
    
    recorder.ondataavailable = async (e) => {
      const reader = new FileReader()
      reader.onloadend = async () => {
        const base64 = reader.result?.toString().split(',')[1]
        const res = await fetch('/functions/v1/transcribe-audio', {
          method: 'POST',
          body: JSON.stringify({ audioBase64: base64 })
        })
        const { transcription } = await res.json()
        onTranscript(transcription)
      }
      reader.readAsDataURL(e.data)
    }
    
    recorder.start(4000) // 4-second chunks
    mediaRecorderRef.current = recorder
    setIsRecording(true)
  }

  const stopRecording = () => {
    mediaRecorderRef.current?.stop()
    setIsRecording(false)
  }

  return (
    <button onClick={isRecording ? stopRecording : startRecording}>
      {isRecording ? 'Stop' : 'Start'} Recording
    </button>
  )
}
LiveTranslation.tsx
import { useEffect, useRef } from 'react'

export function LiveTranslation({ segments }: { segments: Array<{ arabic: string, english: string }> }) {
  const scrollRef = useRef<HTMLDivElement>(null)

  useEffect(() => {
    scrollRef.current?.scrollTo({ top: scrollRef.current.scrollHeight, behavior: 'smooth' })
  }, [segments])

  return (
    <div ref={scrollRef} className="h-screen overflow-y-auto p-6 space-y-4">
      {segments.map((seg, i) => (
        <div key={i} className="border-b pb-4">
          <p className="text-xl text-right mb-2">{seg.arabic}</p>
          <p className="text-lg text-gray-700">{seg.english}</p>
        </div>
      ))}
    </div>
  )
}

function enhanceTranslation(text: string): string {
  // Add (peace be upon him)
  text = text.replace(/(Prophet Muhammad|the Prophet|Rasulullah)/gi, '$1 (peace be upon him)')
  
  // Format Quranic references
  text = text.replace(/Surah (\w+) (\d+):(\d+)/gi, '[Qur'an: Surah $1 $2:$3]')
  
  return text
}
